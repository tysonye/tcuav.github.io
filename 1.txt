3. 数字计数器实现javascriptconst options = {
  useEasing: true,
  useGrouping: true,
  separator: ',',
  decimal: '.',
  suffix: '%'
};

// AI智能识别
const aiCounter = new CountUp('ai-accuracy', 92, options);
// 飞控系统
const flightCounter = new CountUp('flight-control', 90, options);

// 滚动触发
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      aiCounter.start();
      flightCounter.start();
      observer.unobserve(entry.target);
    }
  });
});

document.querySelectorAll('.counter').forEach(el => observer.observe(el));
